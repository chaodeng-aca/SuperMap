transfer_label = function(embedding,k=10,ref_label=cell_type){
  distance = stats::dist(embedding)
  distance = as.matrix(distance)
  distance = distance[1:(nrow(embedding)/2),((nrow(embedding)/2)+1):nrow(embedding)]
  transfered_label = matrix(ncol = 2,nrow = ncol(distance));rownames(transfered_label)=colnames(distance)
  for (cell in colnames(distance)) {
    index = order(distance[,cell])[1:k]
    nearest_cell = rownames(distance)[index]
    freqency = table(ref_label[nearest_cell,])
    transfered_label[cell,1] = names(freqency)[which.max(freqency)]
    transfered_label[cell,2] = max(freqency)/sum(freqency)
  }
  colnames(transfered_label) = c("label","score")
  return(transfered_label)
}

}
