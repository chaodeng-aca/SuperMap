inputdata.10x <- Read10X_h5("pbmc_granulocyte_sorted_10k_filtered_feature_bc_matrix.h5")
rna_counts <- inputdata.10x$`Gene Expression`
atac_counts <- inputdata.10x$Peaks
#-----data proprecessing-----------------------

#RNA modality
pbmc_rna <- CreateSeuratObject(counts = rna_counts,assay = 'RNA')
pbmc_rna = NormalizeData(object = pbmc_rna)
## Normalizing layer: counts
pbmc_rna = FindVariableFeatures(pbmc_rna)
## Finding variable features for layer counts
gene_activity_score = readRDS('gene_activity_score.RDS') #read the gene activity score calculated by Signac
atac_dim = dimReduce(dt1 = atac_counts, K=30)
atac_expression = gene_activity_score
rna = pbmc_rna@assays$RNA$data[rownames(atac_expression),]

cell_label = readRDS('cell_type.RDS')
rna_cluster = cell_label$name
names(rna_cluster) = rownames(cell_label)
atac_cluster = rep(1,ncol(atac_expression))
names(atac_cluster) = colnames(atac_expression)


##run bindsc
bindsc_result <- BiCCA( X = rna,
              Y = t(atac_dim), 
              Z0 = atac_expression, 
              X.clst = rna_cluster,
              Y.clst = atac_cluster,
              alpha = 0.5, 
              lambda = 0.5,
              K = 10,
              temp.path  = "out",
              num.iteration = 100,
              tolerance = 0.01,
              save = TRUE,
              parameter.optimize = FALSE,
              block.size = 0)
